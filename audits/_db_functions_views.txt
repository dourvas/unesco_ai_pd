CREATE OR REPLACE FUNCTION public.cleanup_old_analytics()
 RETURNS void
 LANGUAGE plpgsql
AS $function$
BEGIN
    -- Delete analytics events older than 2 years
    -- (Would first archive to summary table in production)
    DELETE FROM analytics_events 
    WHERE timestamp < NOW() - INTERVAL '2 years';
    
    -- Delete expired IP addresses from consent records (30 days)
    UPDATE consent_records 
    SET ip_address = NULL 
    WHERE granted_at < NOW() - INTERVAL '30 days' 
      AND ip_address IS NOT NULL;
END;
$function$


=== check_tab3_completion body ===
CREATE OR REPLACE FUNCTION public.check_tab3_completion()
 RETURNS trigger
 LANGUAGE plpgsql
AS $function$
BEGIN
    IF NEW.challenge1_completed AND 
       NEW.challenge2_completed AND 
       NEW.challenge3_completed AND 
       NEW.reflection_completed AND
       NOT NEW.activity_completed THEN
        
        NEW.activity_completed = TRUE;
        NEW.activity_completed_at = CURRENT_TIMESTAMP;
    END IF;
    RETURN NEW;
END;
$function$
